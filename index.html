    var today = new Date();
    var month = new GenCal.Month(today.getFullYear(), today.getMonth());
    var el = document.getElementById('gc2');
    el.appendChild(new GenCal.Header(month));
    el.appendChild(new GenCal.Table(month, {
      onEachCell: (month, date, td) => {
        var day = date.getDay();
        if (day == 0 || day == 6)
          td.classList.add('weekend');
      }
    }));
