Day post-processing

var today = new Date();
var month = new GenCal.Month(today.getFullYear(), today.getMonth());
var el = document.getElementById('gc3');
el.appendChild(new GenCal.Header(month));
el.appendChild(new GenCal.Table(month, {
  onEachCell: (month, date, td) => {
    var day = date.getDay();
    if (day == 0 || day == 6)
      td.classList.add('weekend');
  }
}));

Date picker


    var today = new Date();
    var month = new GenCal.Month(today.getFullYear(), today.getMonth());
    var el = document.getElementById('gc4');
    var table = new GenCal.Table(month, {
      onEachCell: (month, date, td) => {
        td.addEventListener('click', () => {
          for (var td_ of table.getElementsByTagName('td')) {
            td_.classList.remove('selected');
          }
          td.classList.add('selected');
        });
      }
    });
    el.appendChild(new GenCal.Header(month));
    el.appendChild(table);
